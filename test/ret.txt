package main
import "fmt"
func main ( ) {
    fmt.Println(Fib(5))
    fmt.Println(Fib(10))
    fmt.Println(Fib(3))
}

func Fib (n int) int{
    
    for true {
        if n > 0
        {
            if n <= 2 {
                return 2
            } else {
                return Fib(n-1) + Fib (n-2)
            }
        } else {
        return 0
        }
    return 0
    }
}
